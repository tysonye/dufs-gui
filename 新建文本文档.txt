    # 根据错误级别设置颜色
    if error:
        color = "#f44336"  # 红色
        level = "错误"
    else:
        color = "#2196f3"  # 蓝色
        level = "信息"
    
    # 添加时间戳和级别
    timestamp = datetime.now().strftime("%H:%M:%S")
    log_entry = f"<span style='color:{color}'>{timestamp} [{level}]</span> {message}"
    
    # 在日志窗口中添加
